<script>

let chart;


// Live WebSocket BTC ticker

const socket = new WebSocket("wss://stream.binance.com:9443/ws/btcusdt@trade");

socket.onmessage = (event)=>{

 const data = JSON.parse(event.data);

 document.getElementById("liveTicker").innerText =

 "Live BTC Price: $" + parseFloat(data.p).toLocaleString();

};


// Main loader

async function loadData(){

 const asset=document.getElementById("asset").value;

 const days=document.getElementById("days").value;


 const market=await fetch(

  `https://api.coingecko.com/api/v3/coins/${asset}?localization=false&market_data=true`

 ).then(r=>r.json());


 const chartData=await fetch(

  `https://api.coingecko.com/api/v3/coins/${asset}/market_chart?vs_currency=usd&days=${days}`

 ).then(r=>r.json());


 const global=await fetch(

  "https://api.coingecko.com/api/v3/global"

 ).then(r=>r.json());


 const lightning=await fetch(

  "https://mempool.space/api/v1/lightning/statistics/latest"

 ).then(r=>r.json());


 const mempool=await fetch(

  "https://mempool.space/api/mempool"

 ).then(r=>r.json());


 document.getElementById("price").innerText=

 "$"+market.market_data.current_price.usd.toLocaleString();


 document.getElementById("cap").innerText=

 "$"+market.market_data.market_cap.usd.toLocaleString();


 document.getElementById("change").innerText=

 market.market_data.price_change_percentage_24h.toFixed(2)+"%";


 document.getElementById("dom").innerText=

 global.data.market_cap_percentage.btc.toFixed(2)+"%";


 document.getElementById("lightning").innerText=

 lightning.total_capacity+" BTC";


 document.getElementById("mempool").innerText=

 mempool.count+" TX";


 const percent=(market.market_data.circulating_supply/21000000)*100;

 document.getElementById("progressBar").style.width=percent+"%";


 const prices=chartData.prices.map(p=>p[1]);

 const labels=chartData.prices.map(p=>new Date(p[0]).toLocaleDateString());


 if(chart) chart.destroy();

 chart=new Chart(document.getElementById("chart"),{

  type:"line",

  data:{

   labels,

   datasets:[{

    data:prices,

    borderColor:"#c8a200",

    fill:true

   }]

  }

 });


 loadETFProxy();

 loadTreasury(market.market_data.current_price.usd);

}


// ETF Proxy Table

async function loadETFProxy(){

 const data=await fetch(

  "https://api.coingecko.com/api/v3/coins/markets?vs_currency=usd&ids=bitcoin,ethereum,solana,tether"

 ).then(r=>r.json());


 const tbody=document.querySelector("#etfTable tbody");

 tbody.innerHTML="";

 data.forEach(asset=>{

  tbody.innerHTML+=`

  <tr>

   <td>${asset.name}</td>

   <td>$${asset.current_price.toLocaleString()}</td>

   <td>$${asset.market_cap.toLocaleString()}</td>

  </tr>`;

 });

}


// Corporate Treasury Demo Data

function loadTreasury(btcPrice){

 const companies=[

  {name:"MicroStrategy",btc:214000},

  {name:"Tesla",btc:9720},

  {name:"Block Inc.",btc:8027}

 ];


 const tbody=document.getElementById("treasuryTable");

 tbody.innerHTML="";

 companies.forEach(c=>{

  tbody.innerHTML+=`

  <tr>

   <td>${c.name}</td>

   <td>${c.btc.toLocaleString()}</td>

   <td>$${(c.btc*btcPrice).toLocaleString()}</td>

  </tr>`;

 });

}

 document.body.classList.toggle("dark");
